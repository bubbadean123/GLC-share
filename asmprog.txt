section .data                           ;Data segment
   userMsg db 'Please enter text up to 255 characters: ' ;Ask the user to enter a number
   lenUserMsg equ $-userMsg             ;The length of the message
   dispMsg db 'You have entered: '
   lenDispMsg equ $-dispMsg                 

section .bss           ;Uninitialized data
   num resb 256
	
section .text          ;Code Segment
   global _start
	
_start:                ;User prompt
   mov eax, 4
   mov ebx, 1
   mov ecx, userMsg
   mov edx, lenUserMsg
   int 80h

   ;Read and store the user input
   mov eax, 3
   mov ebx, 2
   mov ecx, num  
   mov edx, 255          ;255 characters of text max.
   int 80h
	
   ;Output the message 'You have entered: '
   mov eax, 4
   mov ebx, 1
   mov ecx, dispMsg
   mov edx, lenDispMsg
   int 80h  

   ;Output the text entered
   mov eax, 4
   mov ebx, 1
   mov ecx, num
   mov edx, 255
   int 80h
   mov  eax,256      ;number bytes to clear 
   mov  ecx, num     ;ECX will point to the current element to clear
   top:
   mov edx,0
   mov [ecx],edx
   add  ecx,1      ;move pointer to next element
   dec  eax        ;decrement counter
   jnz  top        ;if counter not 0, then loop again

   jmp _start
   ; Exit code
   mov eax, 1
   mov ebx, 0
   int 80h
