sys_write
mov	edx,x		; message length
mov	ecx,x		; message to write
mov	ebx,x		; file descriptor (stdout)
mov	eax,4		; system call number (sys_write)
int	0x80		; call kernel
sys_read:
mov	edx,x		; input buffer length
mov	ecx,x		; input buffer pointer
mov	ebx,x		; file descriptor
mov	eax,3		; system call number (sys_read)
int	0x80		; call kernel
sys_creat
mov	ecx,x		; file permissions
mov	ebx,x	 	; pointer to block of memory containing file name
mov	eax,8		; system call number (sys_creat)
int	0x80		; call kernel  (returns file descriptor/error code in eax register)
sys_open:
mov	edx,x		; file permissionms
mov	ecx,x	        ; file access mode
mov	ebx,x		; pointer to block of memory containing file name
mov	eax,5		; system call number (sys_open)
int	0x80		; call kernel (returns file descriptor/error code in eax register)
sys_exit:
mov ebx, 0		; exit code number		
mov eax, 1		; system call number (sys_exit)
int 80h
FD 1: STDOUT
FD 2: STDIN
